ЁЯТб Basic Solution
def letterCasePermutation(s: str) -> list[str]:
    result = []

    def backtrack(index: int, current: list):
        if index == len(s):
            result.append("".join(current))
            return

        ch = s[index]
        if ch.isdigit():
            # Digits don't branch
            current.append(ch)
            backtrack(index + 1, current)
            current.pop()
        else:
            # Branch: lowercase
            current.append(ch.lower())
            backtrack(index + 1, current)
            current.pop()
            # Branch: uppercase
            current.append(ch.upper())
            backtrack(index + 1, current)
            current.pop()

    backtrack(0, [])
    return result
тЪб Optimized Solution
def letterCasePermutation(s: str) -> list[str]:
    # BFS / iterative approach тАФ avoids recursion overhead
    result = [""]
    for ch in s:
        if ch.isdigit():
            result = [r + ch for r in result]
        else:
            result = [r + c
                      for r in result
                      for c in (ch.lower(), ch.upper())]
    return result

Space (Optimized)O(2с┤╕ ┬╖ n)Stores all partial strings at once in result list

ЁЯТб Basic Solution
def permute(nums: list[int]) -> list[list[int]]:
    result = []
    used = [False] * len(nums)

    def backtrack(current: list):
        if len(current) == len(nums):
            result.append(current[:])  # snapshot
            return
        for i in range(len(nums)):
            if used[i]:
                continue
            used[i] = True
            current.append(nums[i])
            backtrack(current)
            current.pop()       # undo choice
            used[i] = False    # undo mark

    backtrack([])
    return result
тЪб Optimized Solution
def permute(nums: list[int]) -> list[list[int]]:
    # Swap-based: avoid extra `used` array; O(1) extra space per level
    result = []

    def backtrack(start: int):
        if start == len(nums):
            result.append(nums[:])
            return
        for i in range(start, len(nums)):
            nums[start], nums[i] = nums[i], nums[start]  # swap in
            backtrack(start + 1)
            nums[start], nums[i] = nums[i], nums[start]  # swap back

    backtrack(0)
    return result
ЁЯУК Complexity Analysis
MetricValueReason
TimeO(n ┬╖ n!)n! permutations; copying each takes O(n)
Space (Basic)O(n)used[] array + recursion depth n
Space (Optimized)O(n)No extra array; in-place swaps; recursion depth n
Output SpaceO(n ┬╖ n!)Storing all permutations dominates

LeetCode #37 ┬╖ Sudoku Solver

Hard by LeetCode
ЁЯОи Algorithm Visualization
9x9 board тАФ fill each '.' with 1-9

Constraint checks for placing digit d at (r,c):
  тЬТ Row r    : d not in row
  тЬТ Col c    : d not in col
  тЬТ Box 3x3  : d not in box (r//3, c//3)

Backtracking flow:
  find next empty cell ('.')
  try digits 1тАУ9:
    if isValid(r, c, d):
      place d
      recurse()
      if solved тЖТ return True
      undo (board[r][c] = '.')   тЖР backtrack
  return False (trigger backtrack)
ЁЯТб Basic Solution
def solveSudoku(board: list[list[str]]) -> None:
    def is_valid(r, c, d):
        for i in range(9):
            if board[r][i] == d: return False
            if board[i][c] == d: return False
            br, bc = 3*(r//3)+i//3, 3*(c//3)+i%3
            if board[br][bc] == d: return False
        return True

    def backtrack():
        for r in range(9):
            for c in range(9):
                if board[r][c] == '.':
                    for d in "123456789":
                        if is_valid(r, c, d):
                            board[r][c] = d
                            if backtrack(): return True
                            board[r][c] = '.'
                    return False  # no digit worked тЖТ backtrack
        return True  # no empty cell тЖТ solved

    backtrack()
тЪб Optimized Solution
from collections import defaultdict

def solveSudoku(board: list[list[str]]) -> None:
    # Pre-compute candidate sets per cell тАФ prune search space early
    rows = [set() for _ in range(9)]
    cols = [set() for _ in range(9)]
    boxes = [set() for _ in range(9)]

    empty = []
    for r in range(9):
        for c in range(9):
            d = board[r][c]
            if d == '.':
                empty.append((r, c))
            else:
                rows[r].add(d); cols[c].add(d)
                boxes[r//3*3+c//3].add(d)

    def backtrack(idx: int) -> bool:
        if idx == len(empty): return True
        r, c = empty[idx]
        b = r//3*3+c//3
        for d in "123456789":
            if d not in rows[r] | cols[c] | boxes[b]:
                board[r][c] = d
                rows[r].add(d); cols[c].add(d); boxes[b].add(d)
                if backtrack(idx+1): return True
                board[r][c] = '.'
                rows[r].discard(d); cols[c].discard(d); boxes[b].discard(d)
        return False

    backtrack(0)
ЁЯУК Complexity Analysis
MetricValueReason
Time (Basic)O(9^m)m = empty cells; each can try up to 9 digits; worst-case exponential
Time (Optimized)O(9^m)Same worst-case; but set lookups are O(1) тАФ dramatically fewer actual branches
Space (Basic)O(m)Recursion depth = m (max 81)
Space (Optimized)O(81)Three arrays of 9 sets; total cells bounded by 81; O(1) extra beyond output
