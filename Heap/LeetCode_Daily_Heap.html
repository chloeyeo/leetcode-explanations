#703

Kth Largest Element in a Stream

Easy
Algorithm Visualization
  Stream: [4, 5, 8, 2]   k = 3  => 3rd largest

  Min-Heap (size <= k):
  add(4)  heap=[4]
  add(5)  heap=[4,5]
  add(8)  heap=[4,5,8]   (full!)
  add(2)  2 < heap[0]=4  skip

  add(3): push 3 => size=4 > k => pop min(3)
          heap=[4,5,8]

  heap[0] is always the k-th largest!

  Min-Heap (k=3):
       4   <-- heap[0] = answer
      / \
     5   8
Basic Solution
PythonBasic
import heapq

class KthLargest:
    def __init__(self, k, nums):
        self.k = k
        self.heap = []
        for n in nums:
            heapq.heappush(self.heap, n)
            if len(self.heap) > k:
                heapq.heappop(self.heap)

    def add(self, val):
        heapq.heappush(self.heap, val)
        if len(self.heap) > self.k:
            heapq.heappop(self.heap)
        return self.heap[0]
Optimized Solution
PythonOptimized
import heapq

class KthLargest:
    def __init__(self, k, nums):
        self.k = k
        self.heap = nums
        heapq.heapify(self.heap)  # O(n) vs O(n log n)
        while len(self.heap) > k:
            heapq.heappop(self.heap)

    def add(self, val):
        if len(self.heap) < self.k:
            heapq.heappush(self.heap, val)
        elif val > self.heap[0]:
            heapq.heapreplace(self.heap, val)
        return self.heap[0]
Complexity Analysis
VersionTimeSpaceReason
Basic __init__O(n log k)O(k)Push every element + pop if over k
Basic addO(log k)O(1)Single heappush + conditional heappop
Opt __init__O(n)O(k)heapify is O(n); trim is O(n log k)
Opt addO(log k)O(1)heapreplace skips push when val <= min
#215

Kth Largest Element in an Array

Medium
Algorithm Visualization
  nums=[3,2,1,5,6,4]  k=2  => answer: 5

  Min-Heap approach (keep k elements):
  [3]=>[2,3]=>[1,2,3]=>pop1=>[2,3,5]
  =>pop2=>[3,5,6]=>pop3=>[4,5,6]
  heap[0] = 5

  QuickSelect (O(n) avg):
  pivot=4: left=[3,2,1] right=[6,5]
  len(right)=2 = k-1? k=2 => yes? No: store=2, k-1=1
  recurse right=[6,5], k=1
  pivot=5: left=[] right=[6]
  store=1 == k-1=0? No: recurse right=[6], k=1
  pivot=6: store=0 == k-1=0 => return 5!
Basic Solution
PythonMin-Heap O(n log k)
import heapq

class Solution:
    def findKthLargest(self, nums, k):
        heap = []
        for num in nums:
            heapq.heappush(heap, num)
            if len(heap) > k:
                heapq.heappop(heap)
        return heap[0]
Optimized Solution
PythonQuickSelect O(n) avg
import random

class Solution:
    def findKthLargest(self, nums, k):
        def qs(lo, hi):
            pi = random.randint(lo, hi)
            nums[pi], nums[hi] = nums[hi], nums[pi]
            pv = nums[hi]
            st = lo
            for i in range(lo, hi):
                if nums[i] >= pv:
                    nums[st], nums[i] = nums[i], nums[st]
                    st += 1
            nums[st], nums[hi] = nums[hi], nums[st]
            if st == k-1: return nums[st]
            elif st < k-1: return qs(st+1, hi)
            else: return qs(lo, st-1)
        return qs(0, len(nums)-1)
Complexity Analysis
VersionTimeSpaceReason
Min-HeapO(n log k)O(k)Each push/pop O(log k); n elements
QuickSelect avgO(n)O(log n)Expected 2n comparisons; stack depth
QuickSelect worstO(n^2)O(n)Bad pivot every time; random avoids this
#295

Find Median from Data Stream

Hard
Algorithm Visualization
  Two-Heap approach:
  lo = max-heap (negated) = lower half
  hi = min-heap            = upper half

  Stream: 1, 7, 3, 5
  After 1:  lo=[-1]      hi=[]      median=1
  After 7:  lo=[-1]      hi=[7]     median=(1+7)/2=4.0
  After 3:  lo=[-3,-1]   hi=[7]     median=3
  After 5:  lo=[-3,-1]   hi=[5,7]   median=(3+5)/2=4.0

  Invariant:
  |len(lo) - len(hi)| <= 1
  lo[0](negated) <= hi[0] always

  Even total => avg of both tops
  Odd  total => top of bigger heap
Basic Solution
PythonTwo-Heap
import heapq

class MedianFinder:
    def __init__(self):
        self.small = []  # max-heap (negated)
        self.large = []  # min-heap

    def addNum(self, num):
        heapq.heappush(self.small, -num)
        if self.large and -self.small[0] > self.large[0]:
            heapq.heappush(self.large, -heapq.heappop(self.small))
        if len(self.small) > len(self.large) + 1:
            heapq.heappush(self.large, -heapq.heappop(self.small))
        elif len(self.large) > len(self.small):
            heapq.heappush(self.small, -heapq.heappop(self.large))

    def findMedian(self):
        if len(self.small) > len(self.large):
            return float(-self.small[0])
        return (-self.small[0] + self.large[0]) / 2.0
Optimized Solution
PythonOptimized (heappushpop)
import heapq

class MedianFinder:
    def __init__(self):
        self.lo = []  # max-heap (negated) lower half
        self.hi = []  # min-heap upper half

    def addNum(self, num):
        # Route through hi to maintain lo[0] <= hi[0]
        heapq.heappush(self.lo, -heapq.heappushpop(self.hi, num))
        if len(self.lo) > len(self.hi) + 1:
            heapq.heappush(self.hi, -heapq.heappop(self.lo))

    def findMedian(self):
        if len(self.lo) != len(self.hi):
            return float(-self.lo[0])
        return (-self.lo[0] + self.hi[0]) / 2.0
Complexity Analysis
MethodTimeSpaceReason
addNum (basic)O(log n)O(n)Up to 3 heap ops each O(log n)
findMedian (basic)O(1)O(1)Peek at heap tops only
addNum (opt)O(log n)O(n)heappushpop is one atomic operation
findMedian (opt)O(1)O(1)Single compare + peek
