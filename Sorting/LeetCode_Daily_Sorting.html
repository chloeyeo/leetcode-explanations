Basic Solution
class Solution: def sortArray(self, nums): # Merge sort: divide and conquer if len(nums) <= 1: return nums mid = len(nums) // 2 left = self.sortArray(nums[:mid]) right = self.sortArray(nums[mid:]) return self._merge(left, right) def _merge(self, left, right): result, i, j = [], 0, 0 while i < len(left) and j < len(right): if left[i] <= right[j]: result.append(left[i]); i += 1 else: result.append(right[j]); j += 1 result.extend(left[i:]); result.extend(right[j:]) return result
Optimized Solution
class Solution: def sortArray(self, nums): # In-place merge sort (avoids extra slicing) def ms(arr, l, r): if r - l <= 1: return mid = (l + r) // 2 ms(arr, l, mid); ms(arr, mid, r) tmp = arr[l:r] i, j, k = 0, mid - l, l while i < mid - l and j < r - l: if tmp[i] <= tmp[j]: arr[k] = tmp[i]; i += 1 else: arr[k] = tmp[j]; j += 1 k += 1 while i < mid - l: arr[k] = tmp[i]; i += 1; k += 1 ms(nums, 0, len(nums)) return nums

Dutch National Flag: [2,0,2,1,1,0] lo=0 mid=0 hi=5 [2, 0, 2, 1, 1, 0] Step1: nums[mid]=2 swap(mid,hi) hi=4 [0, 0, 2, 1, 1, 2] Step2: nums[mid]=0 swap(lo,mid) lo=1,mid=1 Step3-5: process remaining Final: [0, 0, 1, 1, 2, 2] ✓ Invariant: [0..lo-1]=0s [lo..mid-1]=1s [hi+1..]=2s
Basic Solution
class Solution: def sortColors(self, nums): # Count sort: two passes count = [0, 0, 0] for n in nums: count[n] += 1 idx = 0 for color, freq in enumerate(count): for _ in range(freq): nums[idx] = color; idx += 1
Optimized Solution
class Solution: def sortColors(self, nums): # Dutch National Flag: one pass O(1) space lo, mid, hi = 0, 0, len(nums) - 1 while mid <= hi: if nums[mid] == 0: nums[lo], nums[mid] = nums[mid], nums[lo] lo += 1; mid += 1 elif nums[mid] == 1: mid += 1 else: nums[mid], nums[hi] = nums[hi], nums[mid] hi -= 1
Complexity Analysis

#315 Count of Smaller Numbers After Self
Hard
by LeetCode
Algorithm Visualization
Merge Sort + Index Tracking Input: [5, 2, 6, 1] counts=[0,0,0,0] Indexed: [(5,0),(2,1),(6,2),(1,3)] Divide: [(5,0),(2,1)] [(6,2),(1,3)] Merge [(5,0),(2,1)]: 2<5 => counts[0]+=1 Sorted: [(2,1),(5,0)] Merge [(6,2),(1,3)]: 1<6 => counts[2]+=1 Sorted: [(1,3),(6,2)] Final merge: 1<2 => counts[1]+=1, counts[0]+=1 Result: counts = [2, 1, 1, 0] ✓
Basic Solution
class Solution: def countSmaller(self, nums): # Brute force O(n^2) n, counts = len(nums), [0] * len(nums) for i in range(n): for j in range(i + 1, n): if nums[j] < nums[i]: counts[i] += 1 return counts
Optimized Solution
class Solution: def countSmaller(self, nums): # Merge sort with index tracking: O(n log n) counts = [0] * len(nums) indexed = list(enumerate(nums)) def ms(arr): if len(arr) <= 1: return arr mid = len(arr) // 2 left, right = ms(arr[:mid]), ms(arr[mid:]) merged, i, j = [], 0, 0 while i < len(left) and j < len(right): if left[i][1] > right[j][1]: counts[left[i][0]] += len(right) - j merged.append(left[i]); i += 1 else: merged.append(right[j]); j += 1 merged.extend(left[i:]); merged.extend(right[j:]) return merged ms(indexed) return counts
Complexity Analysis
VersionTimeSpaceReason
BasicO(r²)O(n)Nested loops: each element vs all after it
OptimizedO(n log n)O(n)Merge sort log n levels; crossing-pair count O(n)
