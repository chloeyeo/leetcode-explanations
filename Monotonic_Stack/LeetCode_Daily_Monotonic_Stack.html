#496 Next Greater Element I Easy by LeetCode

💡 Brute Force
Python
from typing import List class Solution: def nextGreaterElement(self, nums1: List[int], nums2: List[int]) -> List[int]: result = [] for num in nums1: found, idx = False, nums2.index(num) for j in range(idx+1, len(nums2)): if nums2[j] > num: result.append(nums2[j]); found=True; break if not found: result.append(-1) return result
⚡ Optimized (Monotonic Stack)
Python
from typing import List class Solution: def nextGreaterElement(self, nums1: List[int], nums2: List[int]) -> List[int]: ng, stk = {}, [] for n in nums2: while stk and stk[-1] < n: ng[stk.pop()] = n stk.append(n) return [ng.get(n, -1) for n in nums1]
📊 Complexity
TypeComplexityReason
Brute TimeO(m*n)Linear scan in nums2 for each nums1 element
Optimal TimeO(m+n)One pass nums2 O(n) + nums1 lookup O(m)
Brute SpaceO(1)No extra space besides result
Optimal SpaceO(n)HashMap + stack hold up to nelements

💡 Brute Force O(n^2)
Python
from typing import List class Solution: def dailyTemperatures(self, temperatures: List[int]) -> List[int]: n = len(temperatures) res = [0] * n for i in range(n): for j in range(i+1, n): if temperatures[j] > temperatures[i]: res[i] = j - i; break return res
⚡ Optimized (Monotonic Stack) O(n)
Python
from typing import List class Solution: def dailyTemperatures(self, temperatures: List[int]) -> List[int]: res, stk = [0]*len(temperatures), [] for i, t in enumerate(temperatures): while stk and temperatures[stk[-1]] < t: p = stk.pop(); res[p] = i - p stk.append(i) return res
📊 Complexity
TypeComplexityReason
Brute TimeO(n^2)Nested loop over each element
Optimal TimeO(n)Each index pushed/popped exactly once
Brute SpaceO(1)No extra space beyond result array
Optimal SpaceO(n)Stack holds up to n indices

💡 Divide and Conquer
Python
from typing import List class Solution: def largestRectangleArea(self, heights: List[int]) -> int: def helper(l, r): if l > r: return 0 m = min(range(l, r+1), key=lambda i: heights[i]) return max(heights[m]*(r-l+1), helper(l, m-1), helper(m+1, r)) return helper(0, len(heights)-1)
⚡ Optimized (Monotonic Stack) O(n)
Python
from typing import List class Solution: def largestRectangleArea(self, heights: List[int]) -> int: mx, stk = 0, [] for i, h in enumerate(heights): s = i while stk and stk[-1][1] > h: idx, ht = stk.pop() mx = max(mx, ht*(i-idx)); s = idx stk.append((s, h)) for idx, ht in stk: mx = max(mx, ht*(len(heights)-idx)) return mx

Optimal TimeO(n)Each bar pushed/popped exactly once
D&C SpaceO(n)Recursion stack depth up to n
Optimal SpaceO(n)Stack holds at most n entries
