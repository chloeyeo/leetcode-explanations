Basic Solution
def climbStairs(self, n: int) -> int:
    # Use dp array of size n+1
    if n <= 2:
        return n
    dp = [0] * (n + 1)
    dp[1] = 1
    dp[2] = 2
    for i in range(3, n + 1):
        dp[i] = dp[i - 1] + dp[i - 2]
    return dp[n]
Optimized Solution (O(1) Space)
def climbStairs(self, n: int) -> int:
    # Rolling variables вҖ” no array needed
    if n <= 2:
        return n
    prev, curr = 1, 2
    for _ in range(3, n + 1):
        prev, curr = curr, prev + curr
    return curr

Basic Solution
def coinChange(self, coins: List[int], amount: int) -> int:
    # dp[i] = min coins to make amount i
    dp = [float('inf')] * (amount + 1)
    dp[0] = 0
    for i in range(1, amount + 1):
        for coin in coins:
            if coin <= i:
                dp[i] = min(dp[i], dp[i - coin] + 1)
    return dp[amount] if dp[amount] != float('inf') else -1
Optimized Solution (Early Exit)
def coinChange(self, coins: List[int], amount: int) -> int:
    # Sort coins descending for cache-friendly access
    dp = [amount + 1] * (amount + 1)
    dp[0] = 0
    coins_sorted = sorted(coins)
    for i in range(1, amount + 1):
        for coin in coins_sorted:
            if coin > i:
                break        # early exit since sorted
            dp[i] = min(dp[i], dp[i - coin] + 1)
    return dp[amount] if dp[amount] <= amount else -1

Basic Solution
def maxCoins(self, nums: List[int]) -> int:
    # Pad with boundary 1s
    nums = [1] + nums + [1]
    n = len(nums)
    dp = [[0] * n for _ in range(n)]

    for length in range(2, n):          # window length
        for left in range(0, n - length):
            right = left + length
            for k in range(left + 1, right):
                coins = (nums[left] * nums[k] * nums[right]
                         + dp[left][k] + dp[k][right])
                dp[left][right] = max(dp[left][right], coins)
    return dp[0][n - 1]
Optimized Solution (Memoization)
def maxCoins(self, nums: List[int]) -> int:
    nums = [1] + nums + [1]
    n = len(nums)
    memo = {}

    def dp(left, right):
        if right - left < 2:
            return 0
        if (left, right) in memo:
            return memo[(left, right)]
        best = 0
        for k in range(left + 1, right):
            coins = (nums[left] * nums[k] * nums[right]
                     + dp(left, k) + dp(k, right))
            best = max(best, coins)
        memo[(left, right)] = best
        return best

    return dp(0, n - 1)
