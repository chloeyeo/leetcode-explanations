Maximum Average Subarray I   (LeetCode #643)
Easy Sliding Window Array
📊 Algorithm Visualization
nums = [1, 12, -5, -6, 50, 3], k = 4 Window slides step by step: Step 1: [1, 12, -5, -6] | sum=2 | avg= 0.500 ^-----------^ Step 2: [12, -5, -6, 50] | sum=51 | avg=12.750 ← max! ^----------^ Step 3: [-5, -6, 50, 3] | sum=42 | avg=10.500 ^-----^ Answer: 12.75000
🐍 Basic Solution
class Solution: def findMaxAverage(self, nums: list[int], k: int) -> float: # Calculate sum of first window window_sum = sum(nums[:k]) max_sum = window_sum # Slide the window: add next, remove first for i in range(k, len(nums)): window_sum += nums[i] - nums[i - k] max_sum = max(max_sum, window_sum) return max_sum / k
⚡ Optimized Solution
class Solution: def findMaxAverage(self, nums: list[int], k: int) -> float: # One-pass with running window, avoid repeated sum() cur = max_s = sum(nums[:k]) for i in range(k, len(nums)): cur += nums[i] - nums[i-k] if cur > max_s: max_s = cur return max_s / k
📈 Complexity Analysis
MetricValueReason
Time (Basic)O(n)Single pass after initial sum
Time (Optimized)O(n)Same, but avoids sum() overhead on first call conceptually cleaner
SpaceO(1)Only 2 variables maintained
Sliding trick+new -oldAdd element entering window, subtract element leaving
Longest Substring Without Repeating Characters   (LeetCode #3)
Medium Sliding Window Hash Map String
📊 Algorithm Visualization
s = "a b c a b c b b" 0 1 2 3 4 5 6 7 l=0: {a:0} r=0 len=1 l=0: {a:0,b:1} r=1 len=2 l=0: {a:0,b:1,c:2} r=2 len=3 ← running max l=1: 'a' repeats → move l to last_a+1=1 {a:3,b:1,c:2} r=3 len=3 l=2: 'b' repeats → move l to last_b+1=2 {a:3,b:4,c:2} r=4 len=3 l=3: 'c' repeats → l to last_c+1=3 {a:3,b:5,c:5} r=5 len=3 l=5: 'b' repeats → l=6 {a:3,b:6,c:5} r=6 len=1 len=1 r=7 len=1 Answer: 3 ("abc")
🐍 Basic Solution
class Solution: def lengthOfLongestSubstring(self, s: str) -> int: seen = set() left = max_len = 0 for right in range(len(s)): # Shrink window until no duplicate while s[right] in seen: seen.remove(s[left]) left += 1 seen.add(s[right]) max_len = max(max_len, right - left + 1) return max_len
⚡ Optimized Solution
class Solution: def lengthOfLongestSubstring(self, s: str) -> int: # Store last seen index → jump left pointer directly (O(1) per step) last_seen = {} left = max_len = 0 for right, ch in enumerate(s): if ch in last_seen and last_seen[ch] >= left: left = last_seen[ch] + 1 last_seen[ch] = right max_len = max(max_len, right - left + 1) return max_len
📈 Complexity Analysis
MetricValueReason
Time (Basic)O(2n) = O(n)Each char can be added/removed from set at most once
Time (Optimized)O(n)One pass; left jumps directly via index map
Space (Basic)O(min(n,σ))σ = charset size (26/128/...)
Space (Optimized)O(min(n,σ))HashMap stores at most σ entries
Key insightleft jumplast_seen[ch]+1 skips redundant shrinks
Minimum Window Substring   (LeetCode #76)
Hard Sliding Window Hash Map Two Pointers
📊 Algorithm Visualization
s="ADOBECODEBANC" t="ABC" need={A:1, B:1, C:1} formed=0 required=3 → Expand right until all chars covered: r:A formed=1 | r:D | r:O | r:B formed=2 r:E | r:C formed=3 ✓ window="ADOBEC" len=6 ← Shrink left while still valid: l:A need[A]↑→formed=2 break window="DOBECODEBA" ... expand right again: Best window tracking: "ADOBEC" len=6 "BANC" len=4 ← MINIMUM ✓ Answer: "BANC"
🐍 Basic Solution
from collections import Counter class Solution: def minWindow(self, s: str, t: str) -> str: if not t or not s: return "" need = Counter(t) window = {} formed = required = len(need) left = 0 best = float('inf'), 0, 0 for right, ch in enumerate(s): window[ch] = window.get(ch, 0) + 1 if ch in need and window[ch] == need[ch]: formed += 1 while formed == required: # Update best window if right - left + 1 < best[0]: best = (right - left + 1, left, right) # Shrink from left lc = s[left] window[lc] -= 1 if lc in need and window[lc] < need[lc]: formed -= 1 left += 1 return "" if best[0] == float('inf') else s[best[1]:best[2]+1]
⚡ Optimized Solution (Filtered String)
from collections import Counter class Solution: def minWindow(self, s: str, t: str) -> str: # Filter s to only chars in t → fewer iterations for large s need = Counter(t) filtered = [(i, ch) for i, ch in enumerate(s) if ch in need] window = {} formed = required = len(need) left = 0 best = float('inf'), 0, 0 for right in range(len(filtered)): i, ch = filtered[right] window[ch] = window.get(ch, 0) + 1 if window[ch] == need[ch]: formed += 1 while formed == required: li, lc = filtered[left] if i - li + 1 < best[0]: best = (i - li + 1, li, i) window[lc] -= 1 if window[lc] < need[lc]: formed -= 1 left += 1 return "" if best[0] == float('inf') else s[best[1]:best[2]+1]
📈 Complexity Analysis
MetricValueReason
Time (Basic)O(|s| + |t|)Each char in s visited at most twice (right+left pointers)
Time (Optimized)O(|s| + |t|)Filtered list reduces iterations when s >> t
SpaceO(|s| + |t|)Filtered list O(|s|), window/need maps O(|t|)
formed checkO(1)Counter equality per unique char only
Key insighttwo-phaseExpand until valid, shrink until invalid, repeat
