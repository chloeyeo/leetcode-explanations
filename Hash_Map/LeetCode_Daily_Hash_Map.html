├────────────┼────────────────┤
│     2      │       0        │  ← found at i=0
└────────────┴────────────────┘

💡 Basic Solution
class Solution:
    def twoSum(self, nums: list[int], target: int) -> list[int]:
        # Brute force O(n²)
        for i in range(len(nums)):
            for j in range(i + 1, len(nums)):
                if nums[i] + nums[j] == target:
                    return [i, j]
        return []
⚡ Optimized Solution
class Solution:
    def twoSum(self, nums: list[int], target: int) -> list[int]:
        # One-pass Hash Map O(n)
        seen = {}  # {value: index}
        for i, num in enumerate(nums):
            need = target - num
            if need in seen:
                return [seen[need], i]
            seen[num] = i
        return []

💡 Basic Solution
from collections import defaultdict

class Solution:
    def groupAnagrams(self, strs: list[str]) -> list[list[str]]:
        # Sort each word as key — O(n * k log k)
        groups = defaultdict(list)
        for word in strs:
            key = tuple(sorted(word))
            groups[key].append(word)
        return list(groups.values())
⚡ Optimized Solution
from collections import defaultdict

class Solution:
    def groupAnagrams(self, strs: list[str]) -> list[list[str]]:
        # Count array as key — O(n * k), avoids sort
        groups = defaultdict(list)
        for word in strs:
            count = [0] * 26
            for c in word:
                count[ord(c) - ord('a')] += 1
            groups[tuple(count)].append(word)
        return list(groups.values())

from collections import Counter

class Solution:
    def findSubstring(self, s: str, words: list[str]) -> list[int]:
        if not s or not words:
            return []
        wl, n, wc = len(words[0]), len(words), Counter(words)
        total = wl * n
        res = []
        # Slide window of total length, step by wl
        for i in range(len(s) - total + 1):
            seen = Counter()
            for j in range(i, i + total, wl):
                w = s[j:j + wl]
                if w in wc:
                    seen[w] += 1
                    if seen[w] > wc[w]:
                        break
                else:
                    break
            else:
                res.append(i)
        return res
⚡ Optimized Solution
from collections import Counter, defaultdict

class Solution:
    def findSubstring(self, s: str, words: list[str]) -> list[int]:
        if not s or not words:
            return []
        wl, n = len(words[0]), len(words)
        need = Counter(words)
        res = []
        # Run wl offset variants to cover all alignments
        for offset in range(wl):
            window = defaultdict(int)
            matched = 0
            left = offset
            for right in range(offset, len(s) - wl + 1, wl):
                w = s[right:right + wl]
                if w in need:
                    window[w] += 1
                    if window[w] == need[w]:
                        matched += 1
                    while window[w] > need[w]:
                        lw = s[left:left + wl]
                        window[lw] -= 1
                        if window[lw] < need[lw]:
                            matched -= 1
                        left += wl
                    if matched == len(need):
                        res.append(left)
                        lw = s[left:left + wl]
                        window[lw] -= 1
                        if window[lw] < need[lw]:
                            matched -= 1
                        left += wl
                else:
                    window.clear()
                    matched = 0
                    left = right + wl
        return res

VersionTimeSpaceReason
BasicO(n·m·wl)O(m)For each of n positions, check m words; m = word count, wl = word length
OptimizedO(n·wl)O(m·wl)Sliding window with wl offset runs; each char visited at most twice per offset — avoids re-scanning inner words
