#933 Number of Recent Calls
by LeetCode
EASY
🎆 ALGORITHM VISUALIZATION
RecentCounter.state: queue = deque() Ping sequence: 1 -> 100 -> 3000 -> 3002 -> 5000 t = 1: queue = [1] Window: [-2999 .. 1] Return len = 1 t = 100: queue = [1, 100] Window: [-2900 .. 100] Return len = 2 t = 3000: queue = [1, 100, 3000] Window: [0 .. 3000] pop left (1 < 0) --> queue = [100, 3000] Return len = 2 t = 3002: queue = [100, 3000, 3002] Window: [2 .. 3002] pop left (100 < 2) --> queue = [3000, 3002] Return len = 2 t = 5000: queue = [3000, 3002, 5000] Window: [2000 .. 5000] Return len = 3 FIFO Window: [================================] ^ ^ oldest newest (t) popleft append
💻 BASIC SOLUTION
# LeetCode #933 -- Number of Recent Calls from collections import deque class RecentCounter: def __init__(self): # Use deque for O(1) append and popleft self.q = deque() def ping(self, t): # Add current timestamp self.q.append(t) # Remove timestamps outside window [t-3000, t] while self.q[0] < t - 3000: self.q.popleft() return len(self.q)
✨ OPTIMIZED

#622 Design Circular Queue
by LeetCode
MEDIUM
🎆 STATE DIAGRAM
Circular Queue (k = 4): Initial: [_][_][_][_] head=0 tail=0 count=0 H/T Enqueue(1): [1][_][_][_] head=0 tail=1 count=1 Enqueue(2): [1][2][_][_] head=0 tail=2 count=2 Enqueue(3): [1][2][3][_] head=0 tail=3 count=3 Enqueue(4): [1][2][3][4] head=0 tail=0 count=4 (FULL) Dequeue(): [_][2][3][4] head=1 tail=0 count=3 Enqueue(5): [5][2][3][4] head=1 tail=1 count=4 (Rewraps!) ^ ^ [slot0] newtail Circular index = (i + 1) % k
💻 BASIC SOLUTION
# LeetCode #622 -- Design Circular Queue class MyCircularQueue: def __init__(self, k): # Fixed-size array as circular buffer self.data = [0] * k self.k = k self.head = 0 self.tail = 0 self.count = 0 def enQueue(self, value): if self.count == self.k: return False self.data[self.tail] = value self.tail = (self.tail + 1) % self.k self.count += 1 return True def deQueue(self): if self.count == 0: return False self.head = (self.head + 1) % self.k self.count -= 1 return True def Front(self): if self.count == 0: return -1 return self.data[self.head] def Rear(self): if self.count == 0: return -1 return self.data[(self.tail - 1) % self.k] def isEmpty(self): return self.count == 0 def isFull(self): return self.count == self.k
✨ OPTIMIZED (k+1 buffer, no counter)
# Optimized: size k+1 buffer distinguishes full vs empty without counter class MyCircularQueue: def __init__(self, k): self.buf = [0] * (k + 1) self.cap = k + 1 self.head = self.tail = 0 def _size(self): return (self.tail - self.head + self.cap) % self.cap def enQueue(self, v): if self._size() == self.cap - 1: return False self.buf[self.tail] = v self.tail = (self.tail + 1) % self.cap return True def deQueue(self): if self._size() == 0: return False self.head = (self.head + 1) % self.cap return True def Front(self): return -1 if self._size() == 0 else self.buf[self.head] def Rear(self): return -1 if self._size() == 0 else self.buf[(self.tail-1)%self.cap] isEmpty = lambda self: self._size() == 0 isFull = lambda self: self._size() == self.cap - 1

#862 Shortest Subarray with Sum at Least K
by LeetCode
HARD
🎆 PREFIX SUM + MONOTONIC DEQUE
nums = [2, -1, 2], K = 3 Prefix sums: p = [0, 2, 1, 3] Monotonic deque (increasing order of prefix sums): i=0: p[0]=0 deque=[0] i=1: p[1]=2 p[1]-p[0] = 2 < 3, no left-pop deque=[0, 1] i=2: p[2]=1 p[2]-p[0] = 1 < 3, no left-pop p[2]=1 < p[1]=2 => pop right 1 (not useful as left bound) deque=[0, 2] i=3: p[3]=3 p[3]-p[0] = 3 >= K => window [0,0], len=3 !! pop left 0; check p[3]-p[2]=2 < 3, stop Best so far: 3 Answer: 3 Key: pop right when p[j] <= p[i] (not useful as left bound) pop left when p[i] - p[deque[0]] >= K (update answer)
💻 BASIC SOLUTION
# LeetCode #862 -- Shortest Subarray with Sum at Least K from collections import deque def shortestSubarray(nums, k): n = len(nums) prefix = [0] * (n + 1) for i in range(n): prefix[i + 1] = prefix[i] + nums[i] dq = deque() res = float("inf") for i in range(n + 1): while dq and prefix[i] - prefix[dq[0]] >= k: res = min(res, i - dq.popleft()) while dq and prefix[dq[-1]] >= prefix[i]: dq.pop() dq.append(i) return res if res != float("inf") else -1
✨ OPTIMIZED (itertools.accumulate)
# Optimized: inline prefix build with itertools.accumulate from itertools import accumulate from collections import deque def shortestSubarray(nums, k): p = [0] + list(accumulate(nums)) dq = deque() res = float("inf") for i, pv in enumerate(p): while dq and pv - p[dq[0]] >= k: res = min(res, i - dq.popleft()) while dq and p[dq[-1]] >= pv: dq.pop() dq.append(i) return res if res != float("inf") else -1
