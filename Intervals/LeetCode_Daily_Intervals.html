#252 Meeting Rooms

by LeetCode
Algorithm Visualization
Input: intervals = [[0,30],[5,10],[15,20]] Sort by start: [0,30] [5,10] [15,20] [0,30] ============================ [5,10] ===== [15,20] ===== [0,30] vs [5,10]: 5 < 30 => OVERLAP! Answer: False Input: intervals = [[7,10],[2,4]] After sort: [[2,4],[7,10]] [2,4] ==== [7,10] ==== 4 <= 7 => No overlap => True
Basic Solution
class Solution: def canAttendMeetings(self, intervals): intervals.sort(key=lambda x: x[0]) for i in range(1, len(intervals)): # overlap if current start < previous end if intervals[i][0] < intervals[i-1][1]: return False return True
Optimized Solution
class Solution: def canAttendMeetings(self, intervals): intervals.sort() return all( intervals[i][0] >= intervals[i-1][1] for i in range(1, len(intervals)) )

#56 Merge Intervals

by LeetCode
Algorithm Visualization
Input: [[1,3],[2,6],[8,10],[15,18]] Sort: [[1,3],[2,6],[8,10],[15,18]] Step 1: merged=[[1,3]] Step 2: 2<=3 => merge => [[1,6]] Step 3: 8>6 => new => [[1,6],[8,10]] Step 4: 15>10=> new => [[1,6],[8,10],[15,18]] Timeline: [1,3] === [2,6] ===== (overlaps => merge) [8,10] === [15,18] ==== Output: [[1,6],[8,10],[15,18]]
Basic Solution
class Solution: def merge(self, intervals): intervals.sort(key=lambda x: x[0]) merged = [] for interval in intervals: if not merged or merged[-1][1] < interval[0]: merged.append(interval) else: merged[-1][1] = max(merged[-1][1], interval[1]) return merged
Optimized Solution
class Solution: def merge(self, intervals): intervals.sort() merged = [intervals[0]] for start, end in intervals[1:]: if start <= merged[-1][1]: merged[-1][1] = max(merged[-1][1], end) else: merged.append([start, end]) return merged
Complexity Analysis
구분복잡도이유
TimeO(n log n)정렬이 병목, 순회는 O(n)
SpaceO(n)merged 결과 배열 저장

#57 Insert Interval

Basic Solution
class Solution: def insert(self, intervals, newInterval): result = [] i, n = 0, len(intervals) # Phase 1: add all before newInterval while i < n and intervals[i][1] < newInterval[0]: result.append(intervals[i]); i += 1 # Phase 2: merge overlapping while i < n and intervals[i][0] <= newInterval[1]: newInterval[0] = min(newInterval[0], intervals[i][0]) newInterval[1] = max(newInterval[1], intervals[i][1]) i += 1 result.append(newInterval) # Phase 3: add remaining while i < n: result.append(intervals[i]); i += 1 return result
Optimized Solution
class Solution: def insert(self, intervals, newInterval): s, e = newInterval left = [iv for iv in intervals if iv[1] < s] right = [iv for iv in intervals if iv[0] > e] overlap = [iv for iv in intervals if iv[0] <= e and iv[1] >= s] if overlap: s = min(s, overlap[0][0]) e = max(e, overlap[-1][1]) return left + [[s, e]] + right
