#643 Maximum Average Subarray I — by LeetCode
Easy
Algorithm Visualization
nums = [1, 12, -5, -6, 50, 3], k=4 Step 1: Initial window [1, 12, -5, -6] sum=2 avg=0.5 |───────── k=4 ───────| Step 2: Slide right [12, -5, -6, 50] sum=51 avg=12.75 |───── k=4 ──────| out=1, in=50 → sum = 2 - 1 + 50 = 51 Step 3: Slide right [-5, -6, 50, 3] sum=42 avg=10.5 |──── k=4 ─────| out=12, in=3 → sum = 51 - 12 + 3 = 42 MAX avg = 12.75 ✓
Basic Solution
Python — Brute Force O(n©k)
class Solution: def findMaxAverage(self, nums: list[int], k: int) -> float: max_avg = float('-inf') for i in range(len(nums) - k + 1): curr_sum = sum(nums[i : i + k]) # O(k) each max_avg = max(max_avg, curr_sum / k) return max_avg
Optimized Solution
Python — Sliding Window O(n)
class Solution: def findMaxAverage(self, nums: list[int], k: int) -> float: window_sum = sum(nums[:k]) # build first window max_sum = window_sum for i in range(k, len(nums)): window_sum += nums[i] - nums[i - k] # add new, remove old max_sum = max(max_sum, window_sum) return max_sum / k
Complexity Analysis
MetricBruteOptimizedReason
TimeO(n©k)O(n)Each element visited once; sum updated in O(1)
SpaceO(1)O(1)Only scalar variables used
#3 Longest Substring Without Repeating Characters — by LeetCode
Medium
Algorithm Visualization
s = "abcabcbb" L R a b c a b c b b |─────────────window→| Expand R until duplicate found: L=0 R=0: {a} len=1 L=0 R=1: {a,b} len=2 L=0 R=2: {a,b,c} len=3 → max=3 L=0 R=3: 'a' seen! → shrink from left L=1 R=3: {b,c,a} len=3 → max=3 F=1 R<4z 'b' seen! → shrink L=2 R=4: {c,a,b} len=3 → max=3 ... Answer = 3 ("abc")
Basic Solution
Python — Set + Two Pointers O(2n)
class Solution: def lengthOfLongestSubstring(self, s: str) -> int: char_set = set() left = 0 max_len = 0 for right in range(len(s)): while s[right] in char_set: # duplicate: shrink left char_set.remove(s[left]) left += 1 char_set.add(s[right]) max_len = max(max_len, right - left + 1) return max_len
Optimized Solution
Python — HashMap (jump left pointer) O(n)
class Solution: def lengthOfLongestSubstring(self, s: str) -> int: char_idx = {} # char → last seen index left = 0 max_len = 0 for right, ch in enumerate(s): if ch in char_idx and char_idx[ch] >= left: left = char_idx[ch] + 1 # jump left past duplicate char_idx[ch] = right max_len = max(max_len, right - left + 1) return max_len
Complexity Analysis
MetricBasicOptimizedReason
TimeO(2n)O(n)Left pointer jumps directly; no inner while loop
SpaceO(min(n,m))O(min(b,m))m=charset size; at most 128 ASCII chars
#76 Minimum Window Substring — by LeetCode
Hard

Basic Solution
Python — Sliding Window with Counter O(|s|+|t|)
from collections import Counter class Solution: def minWindow(self, s: str, t: str) -> str: if not t or not s: return "" need = Counter(t) window = {} formed = required = len(need) left = 0 best = float('inf'), 0, 0 for right, ch in enumerate(s): window[ch] = window.get(ch, 0) + 1 if ch in need and window[ch] == need[ch]: formed -= 1 while formed == 0: # valid window: shrink if right - left + 1 < best[0]: best = right - left + 1, left, right lc = s[left]; left += 1 window[lc] -= 1 if lc in need and window[lc] < need[lc]: formed += 1 return "" if best[0] == float('inf') else s[best[1]:best[2]+1]
Optimized Solution
Python — Filtered string (sparse t) O(|s|+|t|)
from collections import Counter class Solution: def minWindow(self, s: str, t: str) -> str: need = Counter(t) # keep only chars in t → faster for large s, small t filtered = [(i, c) for i, c in enumerate(s) if c in need] window = {} formed = required = len(need) left = 0 best = float('inf'), 0, 0 for right in range(len(filtered)): i, ch = filtered[right] window[ch] = window.get(ch, 0) + 1 if window[ch] == need[ch]: formed -= 1 while formed == 0: j, lc = filtered[left] if i - j + 1 < best[0]: best = i - j + 1, j, i window[lc] -= 1; left += 1 if window[lc] < need[lc]: formed += 1 return "" if best[0] == float('inf') else s[best[1]:best[2]+1]

PracticalSlowerFasterWhen |t| << |s|, filtered skips most iterations
