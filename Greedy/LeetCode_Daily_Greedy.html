LeetCode #435 – Non-overlapping Intervals Medium by LeetCode
📊 Algorithm Visualization
intervals: [[1,2],[2,3],[3,4],[1,3]] sorted by end: [[1,2],[2,3],[1,3],[3,4]] keep [1,2] → prev_end=2 keep [2,3] → prev_end=3 (start==prev_end OK) skip [1,3] → remove! (start 1 < prev_end 3) removed=1 keep [3,4] → prev_end=4 Total removed: 1
🐍 Basic Solution
class Solution: def eraseOverlapIntervals(self, intervals): if not intervals: return 0 intervals.sort(key=lambda x: x[1]) removed = 0 prev_end = intervals[0][1] for i in range(1, len(intervals)): if intervals[i][0] < prev_end: removed += 1 else: prev_end = intervals[i][1] return removed
⚩ Optimized Solution
class Solution: def eraseOverlapIntervals(self, intervals): intervals.sort(key=lambda x: x[1]) removed, prev_end = 0, float('-inf') for start, end in intervals: if start < prev_end: removed += 1 # Keep interval with smaller end (already sorted) else: prev_end = end return removed

import heapq class Solution: def findMaximizedCapital(self, k, w, profits, capital): # Sort projects by capital requirement (ascending) projects = sorted(zip(capital, profits)) max_heap = [] idx = 0 n = len(projects) for _ in range(k): # Push all affordable projects into max-heap (negate for max) while idx < n and projects[idx][0] <= w: heapq.heappush(max_heap, -projects[idx][1]) idx += 1 if not max_heap: break w -= heapq.heappop(max_heap) return w
