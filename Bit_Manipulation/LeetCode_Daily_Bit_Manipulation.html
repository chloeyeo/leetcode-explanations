LeetCode #191 — Number of 1 Bits Easy by LeetCode

💻 Basic Solution (Python)

class Solution:
    def hammingWeight(self, n: int) -> int:
        count = 0
        while n:
            count += n & 1   # check LSB
            n >>= 1           # shift right
        return count

🚀 Optimized Solution (Brian Kernighan's Algorithm)

class Solution:
    def hammingWeight(self, n: int) -> int:
        count = 0
        while n:
            n &= n - 1   # clear lowest set bit: 1100 & 1011 = 1000
            count += 1
        return count

LeetCode #338 — Counting Bits Medium by LeetCode

🎨 Algorithm Visualization

💻 Basic Solution (Python)

class Solution:
    def countBits(self, n: int) -> list[int]:
        result = []
        for i in range(n + 1):
            result.append(bin(i).count('1'))  # O(log i) per call
        return result

🚊 Optimized Solution (DP + Bit Shift)

class Solution:
    def countBits(self, n: int) -> list[int]:
        dp = [0] * (n + 1)
        for i in range(1, n + 1):
            dp[i] = dp[i >> 1] + (i & 1)
            # i>>1 drops LSB → already computed; add back LSB parity
        return dp

💻 Basic Solution — Reverse Bits #190 (Python)

class Solution:
    def reverseBits(self, n: int) -> int:
        result = 0
        for _ in range(32):
            result = (result << 1) | (n & 1)  # take LSB of n → MSB of result
            n >>= 1
        return result

🚀 Optimized Solution (Divide & Conquer Bit Swap)

class Solution:
    def reverseBits(self, n: int) -> int:
        n = (n >> 16) | (n << 16)           # swap 16-bit halves
        n = ((n & 0xFF00FF00) >> 8) | ((n & 0x00FF00FF) << 8)
        n = ((n & 0xF0F0F0F0) >> 4) | ((n & 0x0F0F0F0F) << 4)
        n = ((n & 0xCCCCCCCC) >> 2) | ((n & 0x33333333) << 2)
        n = ((n & 0xAAAAAAAA) >> 1) | ((n & 0x55555555) << 1)
        return n & 0xFFFFFFFF  # mask to 32 bits
