📐 Algorithm Visualization

🐍 Basic Solution (String Conversion)
class Solution: def isPalindrome(self, x: int) -> bool: # Negative numbers are not palindromes if x < 0: return False # Convert to string and compare with reverse s = str(x) return s == s[::-1]
⚡ Optimized Solution (Math — No String)
class Solution: def isPalindrome(self, x: int) -> bool: # Negative or ends in 0 (but not 0 itself) if x < 0 or (x % 10 == 0 and x != 0): return False rev = 0 # Only reverse half the digits while x > rev: rev = rev * 10 + x % 10 x //= 10 # x == rev (even digits) or x == rev//10 (odd digits) return x == rev or x == rev // 10

📐 Algorithm Visualization
Divide without using *, /, % operators dividend = 10, divisor = 3 answer = 3 (10 / 3 = 3.33... → truncate toward 0) Bit-shift doubling approach: ┌────────────────────────────────────┐ │ Start: quotient=0, remain=10 │ │ │ │ 3 × 2⁰= 3 ≤ 10 → take it │ │ 3 × 2¹= 6 ≤ 10 → take it │ │ 3 × 2²= 12 > 10 → stop │ │ │ │ remain = 10 - 6 = 4, q += 2 │ │ 3 × 2⁰= 3 ≤ 4 → take it │ │ 3 × 2¹= 6 > 4 → stop │ │ │ │ remain = 4 - 3 = 1, q += 1 │ │ 3 × 2⁰= 3 > 1 → done │ │ Total quotient = 2 + 1 = 3 ✓ │ └────────────────────────────────────┘
🐍 Basic Solution (Repeated Subtraction)
class Solution: def divide(self, dividend: int, divisor: int) -> int: # Handle overflow edge case INT_MAX = 2**31 - 1 INT_MIN = -2**31 if dividend == INT_MIN and divisor == -1: return INT_MAX sign = -1 if (dividend < 0) ^ (divisor < 0) else 1 a, b = abs(dividend), abs(divisor) quotient = 0 # Repeatedly subtract (TLE for large inputs) while a >= b: a -= b quotient += 1 return sign * quotient
⚡ Optimized Solution (Bit Shifting)
class Solution: def divide(self, dividend: int, divisor: int) -> int: INT_MAX = 2**31 - 1; INT_MIN = -2**31 if dividend == INT_MIN and divisor == -1: return INT_MAX sign = -1 if (dividend < 0) ^ (divisor < 0) else 1 a, b = abs(dividend), abs(divisor) quotient = 0 while a >= b: tmp, mul = b, 1 # Double tmp until it exceeds a while a >= (tmp << 1): tmp <<= 1 mul <<= 1 a -= tmp quotient += mul result = sign * quotient return max(INT_MIN, min(INT_MAX, result))

LeetCode #149 — Max Points on a Line
Hard by LeetCode
📐 Algorithm Visualization
Points: [[1,1],[2,2],[3,3],[1,2],[2,4]] y 4 . * (2,4) 3 . * (3,3) 2 . * (1,2) * (2,2) 1 . * (1,1) +-----------→ x 1 2 3 Slope from (1,1): → to (2,2): dy/dx = 1/1 → slope key (1,1) → to (3,3): dy/dx = 2/2 → key (1,1) same! → to (1,2): dx=0 → vertical line → to (2,4): dy/dx = 3/1 → key (3,1) slope_map from (1,1): {(1,1): 2, vertical: 1, (3,1): 1} max collinear including (1,1) = 2+1 = 3 ← answer!
🐍 Basic Solution (Float Slope)
from collections import defaultdict class Solution: def maxPoints(self, points: list[list[int]]) -> int: n = len(points) if n <= 2: return n res = 2 for i in range(n): slopes = defaultdict(int) for j in range(i+1, n): dx = points[j][0] - points[i][0] dy = points[j][1] - points[i][1] # Use float slope (risk of precision errors) slope = float('inf') if dx == 0 else dy / dx slopes[slope] += 1 res = max(res, max(slopes.values()) + 1) return res
⚡ Optimized Solution (GCD Fraction Key)
from collections import defaultdict from math import gcd class Solution: def maxPoints(self, points: list[list[int]]) -> int: n = len(points) if n <= 2: return n res = 2 for i in range(n): slopes = defaultdict(int) for j in range(i+1, n): dx = points[j][0] - points[i][0] dy = points[j][1] - points[i][1] if dx == 0: key = (0, 1) # vertical line else: # Normalize fraction dy/dx with GCD g = gcd(abs(dx), abs(dy)) dx, dy = dx // g, dy // g # Canonical form: dx always positive if dx < 0: dx, dy = -dx, -dy key = (dx, dy) slopes[key] += 1 res = max(res, max(slopes.values()) + 1) return res
📊 Complexity Analysis
VersionTimeSpaceReason
Basic (float)O(n²) O(n)Double loop; float keys risk hash collision
Optimized (GCD)O(n² log M)O(n)GCD adds O(log M) per pair; exact integer keys
