LeetCode #20 -- Valid Parentheses by LeetCode
Easy

Basic Solution (Python)
class Solution:
    def isValid(self, s: str) -> bool:
        stack = []
        mapping = {')': '(', ']': '[', '}': '{'}
        for char in s:
            if char in mapping:
                top = stack.pop() if stack else '#'
                if mapping[char] != top:
                    return False
            else:
                stack.append(char)
        return not stack
Optimized Solution (Python)
class Solution:
    def isValid(self, s: str) -> bool:
        # Early exit: odd length can never be valid
        if len(s) % 2:
            return False
        stack = []
        close_to_open = {')': '(', ']': '[', '}': '{'}
        for ch in s:
            if ch in close_to_open:
                if not stack or stack[-1] != close_to_open[ch]:
                    return False
                stack.pop()
            else:
                stack.append(ch)
        return not stack

Basic Solution (Python)
class MinStack:
    def __init__(self):
        self.stack = []
        self.min_stack = []

    def push(self, val: int) -> None:
        self.stack.append(val)
        min_val = min(val, self.min_stack[-1] if self.min_stack else val)
        self.min_stack.append(min_val)

    def pop(self) -> None:
        self.stack.pop()
        self.min_stack.pop()

    def top(self) -> int:
        return self.stack[-1]

    def getMin(self) -> int:
        return self.min_stack[-1]
Optimized Solution (Python)
class MinStack:
    # Single stack storing (value, current_min) tuples
    def __init__(self):
        self.stack = []

    def push(self, val: int) -> None:
        cur_min = min(val, self.stack[-1][1]) if self.stack else val
        self.stack.append((val, cur_min))

    def pop(self) -> None:
        self.stack.pop()

    def top(self) -> int:
        return self.stack[-1][0]

    def getMin(self) -> int:
        return self.stack[-1][1]
Complexity Analysis
MetricValueReason
Time (all ops)O(1)push/pop/top/getMin are all single index accesses
Space (basic)O(n)Two stacks each of size n
Space (optimized)O(n)One stack of tuples; better cache locality
LeetCode #84 -- Largest Rectangle in Histogram by LeetCode
Hard

Basic Solution (Python)
class Solution:
    def largestRectangleArea(self, heights: list[int]) -> int:
        n = len(heights)
        max_area = 0
        stack = []
        for i in range(n + 1):
            cur_h = 0 if i == n else heights[i]
            while stack and heights[stack[-1]] > cur_h:
                height = heights[stack.pop()]
                width = i if not stack else i - stack[-1] - 1
                max_area = max(max_area, height * width)
            stack.append(i)
        return max_area
Optimized Solution (Python)
class Solution:
    def largestRectangleArea(self, heights: list[int]) -> int:
        # Sentinel 0s remove edge-case empty-stack checks
        heights = [0] + heights + [0]
        stack = [0]
        max_area = 0
        for i in range(1, len(heights)):
            while heights[i] < heights[stack[-1]]:
                h = heights[stack.pop()]
                w = i - stack[-1] - 1
                max_area = max(max_area, h * w)
            stack.append(i)
        return max_area
Complexity Analysis
MetricValueReason
TimeO(n)Each index pushed and popped at most once
SpaceO(n)Stack holds at most n indices in worst case
Sentinel trickO(1)Avoids separate empty-stack check; cleaner code
