#1408 String Matching in an Array

by LeetCode
Algorithm Visualization
words = ["mass", "as", "hero", "superhero"]

Trie insert (all suffixes):
root
+-- m -> a -> s -> s [END]
+-- a -> s [END]
+-- h -> e -> r -> o [END]
+-- s -> u -> p -> e -> r -> h -> e -> r -> o [END]

"as"   in "mass"?      YES
"hero" in "superhero"? YES
Result: ["as", "hero"]
Basic Solution (Brute Force)
from typing import List class Solution: def stringMatching(self, words: List[str]) -> List[str]: result = [] for i, word in enumerate(words): for j, other in enumerate(words): if i != j and word in other: result.append(word) break return result
Optimized Solution (Trie)
class TrieNode: def __init__(self): self.children = {} self.is_end = False class Solution: def stringMatching(self, words): root = TrieNode() for word in words: for start in range(len(word)): node = root for ch in word[start:]: if ch not in node.children: node.children[ch] = TrieNode() node = node.children[ch] if node.is_end: break node.is_end = True result = [] for word in words: node = root for ch in word: node = node.children[ch] if not node.is_end: result.append(word) return result
Complexity Analysis
TypeComplexityReason
Time (basic)O(n2L)Substring check for each word pair
Time (opt)O(nL2)Insert all suffixes of each word into Trie
Space (opt)O(nL2)Total suffix length stored in Trie
Medium

#208 Implement Trie (Prefix Tree)

by LeetCode
Algorithm Visualization
insert("apple"), insert("app")

root
 +O- a
      +O- p
           +O- p [END:"app"]
                +O- l
                     +O- e [END:"apple"]

search("app")    -> node 'p' is_end=True  -> True
search("ap")     -> node 'p' is_end=False -> False
startsWith("app")-> node exists          -> True
Basic Solution (Dictionary)
class Trie: def __init__(self): self.children = {} self.is_end = False def insert(self, word): node = self for ch in word: if ch not in node.children: node.children[ch] = Trie() node = node.children[ch] node.is_end = True def search(self, word): node = self for ch in word: if ch not in node.children: return False node = node.children[ch] return node.is_end def startsWith(self, prefix): node = self for ch in prefix: if ch not in node.children: return False node = node.children[ch] return True
Optimized Solution (Array-based, O(1) access)
class Trie: def __init__(self): self.children = [None] * 26 self.is_end = False def _idx(self, ch): return ord(ch) - 97 def insert(self, w): n = self for ch in w: i = self._idx(ch) if not n.children[i]: n.children[i] = Trie() n = n.children[i] n.is_end = True def _go(self, s): n = self for ch in s: n = n.children[self._idx(ch)] if not n: return None return n def search(self, w): n = self._go(w); return n is not None and n.is_end def startsWith(self, p): return self._go(p) is not None
Complexity Analysis
TypeComplexityReason
insert / search TimeO(L)Traverse L nodes for word of length L
SpaceO(N*L)N words x avg length L nodes created
Array vs Dict accessO(1)Fixed index (ord-97), no hash overhead
Hard

#212 Word Search II

by LeetCode
Algorithm Visualization
board:               words: ["oath","pea","eat","rain"]
o a a n
e t a e
i h k r
i f l v

Trie: root->o->a->t->h[END:"oath"]
           ->e->a->t[END:"eat"]
           ->p->e->a[END:"pea"]
           ->r->a->i->n[END:"rain"]

DFS from (0,0)='o': o->t->h -> "oath" found!
DFS from (1,0)='e': e->a->t -> "eat"  found!
Basic Solution (DFS per word)
class Solution: def findWords(self, board, words): R, C = len(board), len(board[0]) result = [] def dfs(r, c, i, w): if i == len(w): return True if not (0 <= r < R and 0 <= c < C) or board[r][c] != w[i]: return False tmp, board[r][c] = board[r][c], '#' ok = dfs(r+1,c,i+1,w) or dfs(r-1,c,i+1,w) or dfs(r,c+1,i+1,w) or dfs(r,c-1,i+1,w) board[r][c] = tmp return ok for w in words: if any(dfs(r,c,0,w) for r in range(R) for c in range(C)): result.append(w) return result
Optimized Solution (Trie + DFS + Pruning)
class TrieNode: def __init__(self): self.children = {}; self.word = None class Solution: def findWords(self, board, words): root = TrieNode() for w in words: node = root for ch in w: if ch not in node.children: node.children[ch] = TrieNode() node = node.children[ch] node.word = w R, C = len(board), len(board[0]) result = [] def dfs(r, c, node): ch = board[r][c] if ch not in node.children: return nxt = node.children[ch] if nxt.word: result.append(nxt.word); nxt.word = None board[r][c] = '#' for dr,dc in [(0,1),(0,-1),(1,0),(-1,0)]: nr, nc = r+dr, c+dc if 0 <= nr < R and 0 <= nc < C: dfs(nr,nc,nxt) board[r][c] = ch if not nxt.children: del node.children[ch] for r in range(R): for c in range(C): dfs(r,c,root) return result
Complexity Analysis
TypeComplexityReason
Time (basic)O(W*R*C*4^L)W words x all cells x DFS, L=word length
Time (opt)O(R*C*4^L)Trie searches all words simultaneously
SpaceO(W*L)Total characters stored in Trie
Pruning effectReal speedupDelete empty nodes to shrink future search space
