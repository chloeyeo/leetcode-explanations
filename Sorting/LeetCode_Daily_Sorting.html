Basic Solution (Python)
from typing import List

class Solution:
    def merge(self, nums1: List[int], m: int,
              nums2: List[int], n: int) -> None:
        # Fill then sort - simple but O((m+n)log(m+n))
        for i in range(n):
            nums1[m + i] = nums2[i]
        nums1.sort()
Optimized Solution (Python)
from typing import List

class Solution:
    def merge(self, nums1: List[int], m: int,
              nums2: List[int], n: int) -> None:
        # Three-pointer from end - O(m+n) time, O(1) space
        p1, p2, p = m - 1, n - 1, m + n - 1
        while p1 >= 0 and p2 >= 0:
            if nums1[p1] > nums2[p2]:
                nums1[p] = nums1[p1]; p1 -= 1
            else:
                nums1[p] = nums2[p2]; p2 -= 1
            p -= 1
        while p2 >= 0:
            nums1[p] = nums2[p2]; p2 -= 1; p -= 1

Basic Solution (Python)
from typing import List

class Solution:
    def sortArray(self, nums: List[int]) -> List[int]:
        if len(nums) <= 1:
            return nums
        mid = len(nums) // 2
        left  = self.sortArray(nums[:mid])
        right = self.sortArray(nums[mid:])
        return self._merge(left, right)

    def _merge(self, l, r):
        res, i, j = [], 0, 0
        while i < len(l) and j < len(r):
            if l[i] <= r[j]:
                res.append(l[i]); i += 1
            else:
                res.append(r[j]); j += 1
        return res + l[i:] + r[j:]
Optimized Solution (Python)
from typing import List
import random

class Solution:
    def sortArray(self, nums: List[int]) -> List[int]:
        # Randomized QuickSort - avoids O(n^2) worst case
        def quicksort(lo: int, hi: int) -> None:
            if lo >= hi:
                return
            pivot_idx = random.randint(lo, hi)
            nums[pivot_idx], nums[hi] = nums[hi], nums[pivot_idx]
            pivot, store = nums[hi], lo
            for k in range(lo, hi):
                if nums[k] <= pivot:
                    nums[store], nums[k] = nums[k], nums[store]
                    store += 1
            nums[store], nums[hi] = nums[hi], nums[store]
            quicksort(lo, store - 1)
            quicksort(store + 1, hi)

        quicksort(0, len(nums) - 1)
        return nums

LeetCode #315 — Count of Smaller Numbers After Self
by LeetCode
Hard
Algorithm Visualization
nums = [5, 2, 6, 1]   indices = [0,1,2,3]

Merge Sort on indices, counting cross-inversions:

  Merge [0,1] (vals 5,2) + [2,3] (vals 6,1):
    Compare val[0]=5 vs val[2]=6:
      5 <= 6 => place idx 1 (val=2) first
      ...
    When left element is placed, j elements from
    right side already merged => those j values are
    smaller and appear after current left element.
    count[left_idx] += j

  Final: counts = [2, 1, 1, 0]
  Meaning: 5 has 2 smaller (2,1), 2 has 1 (1), etc.
Basic Solution (Python)
from typing import List

class Solution:
    def countSmaller(self, nums: List[int]) -> List[int]:
        # Brute force O(n^2)
        n = len(nums)
        counts = [0] * n
        for i in range(n):
            for j in range(i + 1, n):
                if nums[j] < nums[i]:
                    counts[i] += 1
        return counts
Optimized Solution (Python)
from typing import List

class Solution:
    def countSmaller(self, nums: List[int]) -> List[int]:
        n = len(nums)
        counts = [0] * n
        indices = list(range(n))

        def merge_sort(idxs: List[int]) -> List[int]:
            if len(idxs) <= 1:
                return idxs
            mid = len(idxs) // 2
            left  = merge_sort(idxs[:mid])
            right = merge_sort(idxs[mid:])
            merged, i, j = [], 0, 0
            while i < len(left) and j < len(right):
                if nums[left[i]] <= nums[right[j]]:
                    # j right elements already placed => smaller
                    counts[left[i]] += j
                    merged.append(left[i]); i += 1
                else:
                    merged.append(right[j]); j += 1
            while i < len(left):
                counts[left[i]] += j
                merged.append(left[i]); i += 1
            merged.extend(right[j:])
            return merged

        merge_sort(indices)
        return counts
