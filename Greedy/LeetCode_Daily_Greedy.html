💡 Basic Solution (Python)

class Solution: def findContentChildren(self, g, s): g.sort() s.sort() child = cookie = 0 while child < len(g) and cookie < len(s): if s[cookie] >= g[child]: child += 1 # child satisfied cookie += 1 return child

⚡ Optimized (Python)

class Solution: def findContentChildren(self, g, s): g.sort(); s.sort() i = j = 0 while i < len(g) and j < len(s): if s[j] >= g[i]: i += 1 j += 1 return i

💡 Basic Solution (Python)

class Solution: def eraseOverlapIntervals(self, intervals): if not intervals: return 0 intervals.sort(key=lambda x: x[1]) removed = 0 prev_end = intervals[0][1] for s, e in intervals[1:]: if s < prev_end: removed += 1 # overlap, remove else: prev_end = e return removed

⚡ Optimized (Python)

class Solution: def eraseOverlapIntervals(self, intervals): intervals.sort(key=lambda x: x[1]) k, prev = 0, float('-inf') for s, e in intervals: if s >= prev: prev = e; k += 1 return len(intervals) - k

Jump Game II (LeetCode #45) Hard — by LeetCode

📊 Algorithm Visualization

nums = [2, 3, 1, 1, 4] idx: 0 1 2 3 4 i=0 far=max(0, 0+2)=2 i==cur(0) jump=1 cur=2 i=1 far=max(2, 1+3)=4 i =2 far=max(4, 2+1)=4 i==cur(2) jump=2 cur=4 cur(4) >= n-1(4) DONE Answer: 2 jumps

💡 Basic Solution (Python)

class Solution: def jump(self, nums): jumps = current_end = farthest = 0 for i in range(len(nums) - 1): farthest = max(farthest, i + nums[i]) if i == current_end: jumps += 1 current_end = farthest if current_end >= len(nums) - 1: break return jumps

⚡ Optimized (Python)

class Solution: def jump(self, nums): jumps = cur = far = 0 for i in range(len(nums) - 1): far = max(far, i + nums[i]) if i == cur: jumps += 1; cur = far return jumps
