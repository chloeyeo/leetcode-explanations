🔵 Basic Solution
from typing import List

class Solution:
    def canAttendMeetings(self, intervals: List[List[int]]) -> bool:
        # Sort intervals by start time
        intervals.sort(key=lambda x: x[0])
        
        # Check each pair of consecutive intervals
        for i in range(1, len(intervals)):
            if intervals[i][0] < intervals[i-1][1]:
                return False  # Overlap found
        
        return True

from typing import List

class Solution:
    def canAttendMeetings(self, intervals: List[List[int]]) -> bool:
        # One-liner: sort and check with all()
        intervals.sort()
        return all(
            intervals[i][0] >= intervals[i-1][1]
            for i in range(1, len(intervals))
        )

🔵 Basic Solution
from typing import List

class Solution:
    def merge(self, intervals: List[List[int]]) -> List[List[int]]:
        # Sort intervals by start time
        intervals.sort(key=lambda x: x[0])
        merged = [intervals[0]]
        
        for start, end in intervals[1:]:
            # If current interval overlaps with last merged
            if start <= merged[-1][1]:
                # Extend the end of last merged interval
                merged[-1][1] = max(merged[-1][1], end)
            else:
                # No overlap, add as new interval
                merged.append([start, end])
        
        return merged
⚡ Optimized Solution
from typing import List

class Solution:
    def merge(self, intervals: List[List[int]]) -> List[List[int]]:
        intervals.sort()
        merged = []
        
        for interval in intervals:
            # Extend last or append new (walrus-style clarity)
            if merged and interval[0] <= merged[-1][1]:
                merged[-1][1] = max(merged[-1][1], interval[1])
            else:
                merged.append(interval[::])  # copy to avoid mutation
        
        return merged

🔵 Basic Solution
from typing import List

# Definition for an Interval.
class Interval:
    def __init__(self, start: int = 0, end: int = 0):
        self.start = start
        self.end = end

class Solution:
    def employeeFreeTime(self,
        schedule: List[List[Interval]]) -> List[Interval]:
        
        # Flatten all intervals from all employees
        all_intervals = []
        for employee in schedule:
            for interval in employee:
                all_intervals.append(interval)
        
        # Sort by start time
        all_intervals.sort(key=lambda x: x.start)
        
        result = []
        cur_end = all_intervals[0].end
        
        for interval in all_intervals[1:]:
            if interval.start > cur_end:
                # Gap found → free time for all employees
                result.append(Interval(cur_end, interval.start))
            cur_end = max(cur_end, interval.end)
        
        return result
⚡ Optimized Solution (Min-Heap)
import heapq
from typing import List

class Solution:
    def employeeFreeTime(self,
        schedule: List[List[Interval]]) -> List[Interval]:
        
        # Min-heap: (start, end, employee_idx, interval_idx)
        heap = []
        for i, employee in enumerate(schedule):
            heapq.heappush(heap, (employee[0].start,
                                  employee[0].end, i, 0))
        
        result = []
        cur_end = heap[0][1]  # earliest start's end
        
        while heap:
            s, e, ei, ii = heapq.heappop(heap)
            if s > cur_end:
                result.append(Interval(cur_end, s))
            cur_end = max(cur_end, e)
            # Push next interval of same employee
            if ii + 1 < len(schedule[ei]):
                nxt = schedule[ei][ii + 1]
                heapq.heappush(heap, (nxt.start, nxt.end, ei, ii+1))
        
        return result
