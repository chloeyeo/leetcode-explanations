#933 Number of Recent Calls — by LeetCode
Easy
💐 Algorithm Visualization
Request stream: ping(1) ping(100) ping(3001) ping(3002) t=1: queue=[1] window [1-3000] count=1 t=100: queue=[1,100] window [1-3000] count=2 t=3001: queue=[1,100,3001] window [1-3000] count=3 t=3002: queue=[100,3001,3002] window[2-3002] ^pop(1) count=3 QUEUE deque: front → [ old ] [ ... ] [ new ] ← back ↖ pop when t - front > 3000
📏 Basic Solution
from collections import deque class RecentCounter: def __init__(self): self.q = deque() # stores timestamps def ping(self, t: int) -> int: self.q.append(t) # add new request while self.q[0] < t - 3000: # remove outside window self.q.popleft() return len(self.q) # remaining = recent calls
⚠矎 Optimized Solution
from collections import deque class RecentCounter: __slots__ = ['q'] def __init__(self): self.q = deque() def ping(self, t: int) -> int: q = self.q q.append(t) cutoff = t - 3000 while q[0] < cutoff: # local var avoids attr lookup q.popleft() return len(q)
📇 Complexity Analysis
OperationTimeReason
append(t)O(1)deque append is amortized O(1)
popleft loopO(1) amortizedeach element enqueued & dequeued once
len(q)O(1)deque tracks length internally
SpaceO(W)W = max requests in 3000ms window
#649 Dota2 Senate — by LeetCode
Medium
💐 Algorithm Visualization
senate = "RDD" Radiant queue: [0] Dire queue: [1, 2] Round 1: R(0) vs D(1): R wins → R rejoins as [3], D eliminated D(2) vs R(3): D wins → D rejoins as [4], R eliminated Radiant queue empty → "Dire" wins senate = "RD" R queue: [0] D queue: [1] R(0) vs D(1): R bans D → R rejoins [2] D queue empty -→ "Radiant" wins KEY: index comparison tells who goes first in current round
📏 Basic Solution
from collections import deque class Solution: def predictPartyVictory(self, senate: str) -> str: radiant = deque() dire = deque() n = len(senate) for i, s in enumerate(senate): if s == 'R': radiant.append(i) else: dire.append(i) while radiant and dire: r = radiant.popleft() d = dire.popleft() if r < d: # R acts first, bans D radiant.append(r + n) # R gets next-round turn else: # D acts first, bans R dire.append(d + n) return "Radiant" if radiant else "Dire"
⚠矎 Optimized Solution
from collections import deque class Solution: def predictPartyVictory(self, senate: str) -> str: R, D = deque(), deque() for i, c in enumerate(senate): (R if c == 'R' else D).append(i) n = len(senate) while R and D: r, d = R.popleft(), D.popleft() (R if r < d else D).append((r if r < d else d) + n) return "Radiant" if R else "Dire"
📇 Complexity Analysis
OperationTimeReason
Build queuesO(n)single pass through senate string
While loopO(n)each senator processed once per round; at most n rounds
TotalO(n)each senator eliminated exactly once overall
SpaceO(n)two deques hold all n senators
#862 Shortest Subarray with Sum at Least K — by LeetCode
Hard
💐 Algorithm Visualization
nums = [2, -1, 2], k = 3 prefix = [0, 2, 1, 3] Monotonic deque (stores indices of increasing prefix sums): i=0: prefix[0]=0 deque=[0] i=1: prefix[1]=2 deque=[0,1] prefix[1]-prefix[0]=2 < 3 i=2: prefix[2]=1 2>1 → pop(1) deque=[0,2] i=3: prefix[3]=3 prefix[3]-prefix[0]=3 ∥ 3 → len=3-0=3 ✓ pop(0), check next... deque=[2] prefix[3]-prefix[2]=2 < 3 stop Result: 3 DEQUE maintains: indices with increasing prefix sums Pop back when new prefix is smaller (useless future candidate) Pop front when sum condition met (found valid subarray)
📏 Basic Solution
from collections import deque from typing import List class Soluution: def shortestSubarray(self, nums: List[int], k: int) -> int: n = len(nums) prefix = [0] * (n + 1) for i in range(n): # build prefix sums prefix[i+1] = prefix[i] + nums[i] res = float('inf') dq = deque() # monotonic increasing deque for j in range(n + 1): # pop back: if prefix[j] <= prefix[dr[-1]], dq[-1] useless while dq and prefix[j] <= prefix[dq[-1]]: dq.pop() # pop front: valid subarrays ending at j while dq and prefix[j] - prefix[dq[0]] >= k: res = min(res, j - dq.popleft()) dq.append(j) return res if res != float('inf') else -1
⚠矎 Optimized Solution
from collections import deque from itertools import accumulate from typing import List class Solution: def shortestSubarray(self, nums: List[int], k: int) -> int: prefix = [0] + list(accumulate(nums)) # O(n), cleaner res, dq = float('inf'), deque() for j, pj in enumerate(prefix): while dq and pj <= prefix[dr[-1]]: dq.pop() # maintain monotone while dq and pj - prefix[dq[0]] >= k: res = min(res, j - dq.popleft()) # update answer dq.append(j) return -1 if res == float('inf') else res

OperationTimeReason
Build prefixO(n)single accumulate pass
Main loopO(n)each index pushed/popped from deque at most once
pop backO(1) amortizedeach element removed at most once total
pop frontO(1) amortizedeach element removed at most once total
