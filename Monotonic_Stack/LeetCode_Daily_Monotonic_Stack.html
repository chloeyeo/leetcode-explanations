class MyQueue:
    def __init__(self):
        self.stack_in = []
        self.stack_out = []

    def push(self, x: int) -> None:
        self.stack_in.append(x)

    def pop(self) -> int:
        if not self.stack_out:
            while self.stack_in:
                self.stack_out.append(self.stack_in.pop())
        return self.stack_out.pop()

    def peek(self) -> int:
        if not self.stack_out:
            while self.stack_in:
                self.stack_out.append(self.stack_in.pop())
        return self.stack_out[-1]

    def empty(self) -> bool:
        return not self.stack_in and not self.stack_out

from typing import List

class Solution:
    def dailyTemperatures(self, temperatures: List[int]) -> List[int]:
        n = len(temperatures)
        ans = [0] * n
        stack = []  # indices of unresolved days

        for i in range(n):
            # pop all days colder than today
            while stack and temperatures[i] > temperatures[stack[-1]]:
                prev = stack.pop()
                ans[prev] = i - prev
            stack.append(i)

        return ans

class Solution:
    def dailyTemperatures(self, temperatures: List[int]) -> List[int]:
        n = len(temperatures)
        ans = [0] * n
        stack = []

        # Iterate in reverse ŌĆö avoid extra stack operations
        for i in range(n - 1, -1, -1):
            j = i + 1
            while j < n and temperatures[j] <= temperatures[i]:
                if ans[j] == 0:
                    j = n  # no warmer day found
                    break
                j += ans[j]  # jump to next warmer
            if j < n:
                ans[i] = j - i
        return ans

class Solution:
    def largestRectangleArea(self, heights: List[int]) -> int:
        stack = []  # monotonic increasing stack of indices
        max_area = 0
        heights = heights + [0]  # sentinel to flush stack at end

        for i, h in enumerate(heights):
            start = i
            while stack and stack[-1][1] > h:
                idx, height = stack.pop()
                max_area = max(max_area, height * (i - idx))
                start = idx
            stack.append((start, h))

        return max_area

class Solution:
    def largestRectangleArea(self, heights: List[int]) -> int:
        n = len(heights)
        stack = [-1]  # sentinel index
        max_area = 0

        for i in range(n + 1):
            h = heights[i] if i < n else 0
            while stack[-1] != -1 and heights[stack[-1]] >= h:
                height = heights[stack.pop()]
                width  = i - stack[-1] - 1
                max_area = max(max_area, height * width)
            stack.append(i)

        return max_area
