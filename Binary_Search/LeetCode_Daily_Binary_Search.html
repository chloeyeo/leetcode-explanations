class Solution: def findMedianSortedArrays(self, A: list[int], B: list[int]) -> float: if len(A) > len(B): A, B = B, A m, n = len(A), len(B) INF = float('inf') lo, hi = 0, m while lo <= hi: i = (lo + hi) // 2 j = (m + n + 1) // 2 - i # sentinel к°ұлӘ°лӢЁ к·ёк°ң мјҳШё” Aleft = A[i-1] if i else -INF; Aright = A[i] if i < m else INF Bleft = B[j-1] if j else -INF; Bright = B[j] if j < n else INF if Aleft > Bright: hi = i - 1 elif Bleft > Aright: lo = i + 1 else: lmax = max(Aleft, Bleft) return lmax if (m+n)%2 else (lmax + min(Aright, Bright)) / 2
