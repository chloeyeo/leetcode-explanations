Basic Solution (Iterative)
def reverseList(self, head: Optional[ListNode]) -> Optional[ListNode]:
    prev = None
    curr = head

    while curr:
        nxt       = curr.next   # save next
        curr.next = prev       # reverse pointer
        prev      = curr       # advance prev
        curr      = nxt        # advance curr

    return prev               # prev is new head
Optimized Solution (Recursive)
def reverseList(self, head: Optional[ListNode]) -> Optional[ListNode]:
    if not head or not head.next:
        return head            # base case: 0 or 1 node

    new_head = self.reverseList(head.next)
    head.next.next = head     # reverse pointer at tail end
    head.next      = None    # break original link
    return new_head
Complexity Analysis

#19 Remove Nth Node From End of List Medium by LeetCode
Algorithm Visualization
Remove 2nd node from end of: 1 -> 2 -> 3 -> 4 -> 5

Use two pointers gap = n+1 = 3:

  ^fast (advance n+1 steps first)
  ^slow

After advancing fast n+1=3 steps:
dummy -> 1 -> 2 -> 3 -> 4 -> 5
               ^slow       ^fast (at None?)

Actually: fast points to node at index 3 (1-based)
Move both until fast == None:

dummy -> 1 -> 2 -> 3 -> 4 -> 5
               ^slow            ^fast=None

slow.next = slow.next.next  =>  skip node 4

Result: 1 -> 2 -> 3 -> 5
Basic Solution (Two-Pass)
def removeNthFromEnd(self, head, n):
    # Pass 1: count length
    length = 0
    node = head
    while node:
        length += 1
        node = node.next

    # Pass 2: reach (length - n - 1)-th node
    dummy = ListNode(0, head)
    node = dummy
    for _ in range(length - n):
        node = node.next
    node.next = node.next.next
    return dummy.next
Optimized Solution (One-Pass, Two Pointers)
def removeNthFromEnd(self, head, n):
    dummy = ListNode(0, head)
    fast = slow = dummy

    # advance fast by n+1 steps to maintain gap
    for _ in range(n + 1):
        fast = fast.next

    # move both until fast reaches end
    while fast:
        fast = fast.next
        slow = slow.next

    # slow is just before the target node
    slow.next = slow.next.next
    return dummy.next
Complexity Analysis
VersionTimeSpaceReason
Two-PassO(n)O(1)Two full traversals of the list
One-PassO(n)O(1)Single traversal; constant extra space
#25 Reverse Nodes in k-Group Hard by LeetCode
Algorithm Visualization
Input: 1 -> 2 -> 3 -> 4 -> 5,  k = 2

Group 1: [1, 2]  =>  reverse  =>  2 -> 1
Group 2: [3, 4]  =>  reverse  =>  4 -> 3
Group 3: [5]     =>  less than k, keep as-is  =>  5

Result: 2 -> 1 -> 4 -> 3 -> 5

Step-by-step for Group 1:
  dummy -> 1 -> 2 -> 3 -> ...
  groupPrev = dummy

  Check k nodes exist: 1,2 ✓
  kth node = 2, groupNext = 3

  Reverse 1..2 in-place:
    None <- 1 <- 2   (3 disconnected temporarily)
  
  Connect: dummy -> 2 -> 1 -> 3 -> ...
           groupPrev = node 1  (tail of reversed group)
  
  Repeat for next group starting at 3...
Basic Solution
def reverseKGroup(self, head, k):
    def getKth(node, k):
        while node and k > 0:
            node = node.next
            k -= 1
        return node

    dummy = ListNode(0)
    dummy.next = head
    group_prev = dummy

    while True:
        kth = getKth(group_prev, k)
        if not kth:              # fewer than k nodes left
            break

        group_next = kth.next
        # reverse nodes from group_prev.next to kth
        prev, curr = group_next, group_prev.next
        while curr != group_next:
            tmp        = curr.next
            curr.next  = prev
            prev       = curr
            curr       = tmp

        # reconnect
        tmp              = group_prev.next  # old head = new tail
        group_prev.next  = kth             # connect to new head
        group_prev       = tmp             # advance group_prev

    return dummy.next
Optimized Solution (Recursive)
def reverseKGroup(self, head, k):
    # check if k nodes remain
    node, count = head, 0
    while node and count < k:
        node = node.next
        count += 1
    if count < k:
        return head          # base case: remaining < k

    # reverse current k-group
    prev, curr = None, head
    for _ in range(k):
        nxt        = curr.next
        curr.next  = prev
        prev       = curr
        curr       = nxt

    # head is now tail; attach result of rest
    head.next = self.reverseKGroup(curr, k)
    return prev           # prev is new head
Complexity Analysis
VersionTimeSpaceReason
IterativeO(n)O(1)Each node visited at most twice; O(1) pointers
RecursiveO(n)O(n/k)Recursion depth = number of groups = n/k
