Basic Solution
class Solution:
    def canAttendMeetings(self, intervals: list[list[int]]) -> bool:
        intervals.sort(key=lambda x: x[0])
        for i in range(1, len(intervals)):
            # previous end > current start means overlap
            if intervals[i - 1][1] > intervals[i][0]:
                return False
        return True
Optimized Solution
class Solution:
    def canAttendMeetings(self, intervals: list[list[int]]) -> bool:
        intervals.sort(key=lambda x: x[0])
        return not any(
            a[1] > b[0] for a, b in zip(intervals, intervals[1:])
        )

Basic Solution
class Solution:
    def merge(self, intervals: list[list[int]]) -> list[list[int]]:
        if not intervals:
            return []
        intervals.sort(key=lambda x: x[0])
        merged = [intervals[0]]
        for start, end in intervals[1:]:
            if start <= merged[-1][1]:
                merged[-1][1] = max(merged[-1][1], end)
            else:
                merged.append([start, end])
        return merged
Optimized Solution
from functools import reduce

class Solution:
    def merge(self, intervals: list[list[int]]) -> list[list[int]]:
        intervals.sort(key=lambda x: x[0])
        def merge_two(acc, cur):
            if acc and cur[0] <= acc[-1][1]:
                acc[-1][1] = max(acc[-1][1], cur[1])
            else:
                acc.append(cur)
            return acc
        return reduce(merge_two, intervals, [])

LeetCode #57 - Insert Interval Hard by LeetCode

Basic Solution
class Solution:
    def insert(self, intervals: list[list[int]], newInterval: list[int]) -> list[list[int]]:
        result, i, n = [], 0, len(intervals)
        # Part 1: intervals ending before new starts
        while i < n and intervals[i][1] < newInterval[0]:
            result.append(intervals[i]); i += 1
        # Part 2: overlapping - merge into newInterval
        while i < n and intervals[i][0] <= newInterval[1]:
            newInterval[0] = min(newInterval[0], intervals[i][0])
            newInterval[1] = max(newInterval[1], intervals[i][1])
            i += 1
        result.append(newInterval)
        # Part 3: remaining right intervals
        while i < n:
            result.append(intervals[i]); i += 1
        return result
Optimized Solution
import bisect

class Solution:
    def insert(self, intervals: list[list[int]], newInterval: list[int]) -> list[list[int]]:
        starts = [iv[0] for iv in intervals]
        ends   = [iv[1] for iv in intervals]
        # left boundary: first interval whose end >= newInterval start
        left  = bisect.bisect_left(ends, newInterval[0])
        # right boundary: first interval whose start > newInterval end
        right = bisect.bisect_right(starts, newInterval[1])
        if left < right:
            newInterval = [
                min(newInterval[0], intervals[left][0]),
                max(newInterval[1], intervals[right - 1][1])
            ]
        return intervals[:left] + [newInterval] + intervals[right:]
