💻 Basic Solution (Python)
def findContentChildren(self, g: List[int], s: List[int]) -> int:
    g.sort()
    s.sort()
    child = 0
    cookie = 0
    while child < len(g) and cookie < len(s):
        if s[cookie] >= g[child]:
            child += 1
        cookie += 1
    return child
⚡ Optimized Solution (Python)
import bisect

def findContentChildren(self, g: List[int], s: List[int]) -> int:
    g.sort()
    s.sort()
    count = 0
    si = 0
    for greed in g:
        idx = bisect.bisect_left(s, greed, si)
        if idx < len(s):
            count += 1
            si = idx + 1
    return count

💻 Basic Solution (Python)
def eraseOverlapIntervals(self, intervals: List[List[int]]) -> int:
    if not intervals:
        return 0
    intervals.sort(key=lambda x: x[1])
    remove = 0
    end = intervals[0][1]
    for i in range(1, len(intervals)):
        if intervals[i][0] < end:
            remove += 1
        else:
            end = intervals[i][1]
    return remove
⚡ Optimized Solution (Python)
def eraseOverlapIntervals(self, intervals: List[List[int]]) -> int:
    intervals.sort(key=lambda x: x[1])
    keep = 1
    end = intervals[0][1]
    for start, e in intervals[1:]:
        if start >= end:
            keep += 1
            end = e
    return len(intervals) - keep
📊 Complexity Analysis
VersionTimeSpaceReason
BasicO(n log n)O(1)Sort by end time, greedy keep non-overlapping
OptimizedO(n log n)O(1)Count kept intervals, subtract from total — cleaner logic

💻 Basic Solution (Python)
def minPatches(self, nums: List[int], n: int) -> int:
    miss = 1   # smallest missing sum we cannot form
    patches = 0
    i = 0
    while miss <= n:
        if i < len(nums) and nums[i] <= miss:
            miss += nums[i]
            i += 1
        else:
            # patch with 'miss' itself to double coverage
            miss += miss
            patches += 1
    return patches
⚡ Optimized Solution (Python)
def minPatches(self, nums: List[int], n: int) -> int:
    miss, patches, i = 1, 0, 0
    while miss <= n:
        if i < len(nums) and nums[i] <= miss:
            miss += nums[i]; i += 1
        else:
            miss <<= 1   # miss *= 2 using bit shift
            patches += 1
    return patches
📊 Complexity Analysis
VersionTimeSpaceReason
BasicO(m + log n)O(1)m = len(nums); patches double miss so at most log n patches
OptimizedO(m + log n)O(1)Bit shift (<<= 1) replaces multiply — minor constant speedup
